package com.example.kreatorpostacirpg;

import android.os.Bundle;
import android.view.View;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.EditText;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.Spinner;
import android.widget.TextView;
import android.widget.Toast;

import androidx.appcompat.app.AppCompatActivity;

import java.util.ArrayList;

public class MainActivity extends AppCompatActivity {

    private EditText editImie;
    private RadioGroup radioGroupPlec;
    private Spinner spinnerRasa;
    private CheckBox checkWlamywanie, checkAlchemia, checkKowalstwo;
    private Button buttonStworz, buttonResetuj;
    private TextView textPodsumowanie, textDruzyna;

    private ArrayList<Postac> druzyna = new ArrayList<>();

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        editImie = findViewById(R.id.editImie);
        radioGroupPlec = findViewById(R.id.radioGroupPlec);
        spinnerRasa = findViewById(R.id.spinnerRasa);

        checkWlamywanie = findViewById(R.id.checkWlamywanie);
        checkAlchemia = findViewById(R.id.checkAlchemia);
        checkKowalstwo = findViewById(R.id.checkKowalstwo);

        buttonStworz = findViewById(R.id.buttonStworz);
        buttonResetuj = findViewById(R.id.buttonResetuj);

        textPodsumowanie = findViewById(R.id.textPodsumowanie);
        textDruzyna = findViewById(R.id.textDruzyna);

        // Spinner
        ArrayAdapter<String> adapter = new ArrayAdapter<>(
                this,
                android.R.layout.simple_spinner_item,
                getResources().getStringArray(R.array.rasy_array)
        );
        adapter.setDropDownViewResource(android.R.layout.simple_spinner_dropdown_item);
        spinnerRasa.setAdapter(adapter);

        // LIMIT 2 umiejętności
        View.OnClickListener limitListener = new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                int ile = policzZaznaczone();
                CheckBox box = (CheckBox) v;

                // jeśli kliknięciem przekroczono limit → cofamy
                if (ile > 2) {
                    box.setChecked(false);
                    Toast.makeText(MainActivity.this,
                            "Możesz wybrać maksymalnie 2 umiejętności!",
                            Toast.LENGTH_SHORT).show();
                }
            }
        };

        checkWlamywanie.setOnClickListener(limitListener);
        checkAlchemia.setOnClickListener(limitListener);
        checkKowalstwo.setOnClickListener(limitListener);

        // Stwórz postać
        buttonStworz.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {

                String imie = editImie.getText().toString().trim();
                if (imie.isEmpty()) {
                    Toast.makeText(MainActivity.this, "Wprowadź imię postaci", Toast.LENGTH_SHORT).show();
                    return;
                }

                int selectedId = radioGroupPlec.getCheckedRadioButtonId();
                if (selectedId == -1) {
                    Toast.makeText(MainActivity.this, "Wybierz płeć postaci", Toast.LENGTH_SHORT).show();
                    return;
                }
                RadioButton rb = findViewById(selectedId);
                String plec = rb.getText().toString();

                String rasa = spinnerRasa.getSelectedItem().toString();

                ArrayList<String> umiejetnosci = new ArrayList<>();
                if (checkWlamywanie.isChecked()) umiejetnosci.add("Włamywanie");
                if (checkAlchemia.isChecked()) umiejetnosci.add("Alchemia");
                if (checkKowalstwo.isChecked()) umiejetnosci.add("Kowalstwo");

                Postac postac = new Postac(imie, plec, rasa, umiejetnosci);

                // dodanie do drużyny
                druzyna.add(postac);
                textDruzyna.setText("Aktualna wielkość drużyny: " + druzyna.size());

                // podsumowanie
                textPodsumowanie.setText(postac.generujPodsumowanie());
            }
        });

        // Reset
        buttonResetuj.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                editImie.setText("");
                radioGroupPlec.clearCheck();
                spinnerRasa.setSelection(0);

                checkWlamywanie.setChecked(false);
                checkAlchemia.setChecked(false);
                checkKowalstwo.setChecked(false);

                textPodsumowanie.setText("");
            }
        });
    }

    private int policzZaznaczone() {
        int count = 0;
        if (checkWlamywanie.isChecked()) count++;
        if (checkAlchemia.isChecked()) count++;
        if (checkKowalstwo.isChecked()) count++;
        return count;
    }
}

package com.example.kreatorpostacirpg;

import java.util.ArrayList;

public class Postac {

    private String imie;
    private String plec;
    private String rasa;
    private ArrayList<String> umiejetnosci;

    private int sila;
    private int zrecznosc;

    public Postac(String imie, String plec, String rasa, ArrayList<String> umiejetnosci) {
        this.imie = imie;
        this.plec = plec;
        this.rasa = rasa;
        this.umiejetnosci = umiejetnosci;

        ustawBonusyRasowe();
    }

    private void ustawBonusyRasowe() {
        switch (rasa) {
            case "Krasnolud":
                sila = 12;
                zrecznosc = 8;
                break;
            case "Elf":
                sila = 8;
                zrecznosc = 12;
                break;
            case "Ork":
                sila = 14;
                zrecznosc = 6;
                break;
            default: // Człowiek
                sila = 10;
                zrecznosc = 10;
                break;
        }
    }

    public String generujPodsumowanie() {
        StringBuilder sb = new StringBuilder();
        sb.append("Imię: ").append(imie).append("\n");
        sb.append("Płeć: ").append(plec).append("\n");
        sb.append("Rasa: ").append(rasa).append("\n");
        sb.append("Siła: ").append(sila).append("\n");
        sb.append("Zręczność: ").append(zrecznosc).append("\n");
        sb.append("Umiejętności: ");

        if (umiejetnosci.isEmpty()) {
            sb.append("brak");
        } else {
            sb.append(String.join(", ", umiejetnosci));
        }

        return sb.toString();
    }
}
