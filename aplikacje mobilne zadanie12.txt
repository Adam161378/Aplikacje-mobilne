package com.example.moja_aplikacja;

import android.os.Bundle;
import android.view.View;
import android.widget.Button;
import android.widget.CheckBox;
import android.widget.EditText;
import android.widget.RadioButton;
import android.widget.RadioGroup;
import android.widget.Switch;
import android.widget.TextView;
import android.widget.Toast;

import androidx.activity.EdgeToEdge;
import androidx.appcompat.app.AppCompatActivity;
import androidx.core.graphics.Insets;
import androidx.core.view.ViewCompat;
import androidx.core.view.WindowInsetsCompat;

import java.util.ArrayList;

public class MainActivity extends AppCompatActivity {

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        EdgeToEdge.enable(this);
        setContentView(R.layout.activity_main);
        ViewCompat.setOnApplyWindowInsetsListener(findViewById(R.id.main), (v, insets) -> {
            Insets systemBars = insets.getInsets(WindowInsetsCompat.Type.systemBars());
            v.setPadding(systemBars.left, systemBars.top, systemBars.right, systemBars.bottom);
            return insets;
        });

        CheckBox checkboxSer = findViewById(R.id.checkboxSer);
        CheckBox checkboxSzynka = findViewById(R.id.checkboxMieso);
        CheckBox checkboxPieczarki = findViewById(R.id.checkboxGrzyby);
        CheckBox chechBoxznizka = findViewById(R.id.checkBoxznizka);

        checkboxSer.setOnClickListener(v -> {
            aktualizujCene(); // Wywołujemy naszą główną metodę
        });

        checkboxSzynka.setOnClickListener(v -> {
            aktualizujCene(); // Wywołujemy naszą główną metodę
        });

        checkboxPieczarki.setOnClickListener(v -> {
            aktualizujCene(); // Wywołujemy naszą główną metodę
        });

        chechBoxznizka.setOnClickListener(v -> {
            if (chechBoxznizka.isChecked()) {
                Toast.makeText(this, "Znizka wlaczona", Toast.LENGTH_SHORT).show();
            }else{
                Toast.makeText(this, "Znizka wylaczona", Toast.LENGTH_SHORT).show();
            }

        });

        RadioGroup radioGroupRozmiar = findViewById(R.id.radioGroupRozmiar);
        radioGroupRozmiar.setOnCheckedChangeListener((group, checkedId) -> {
            aktualizujCene(); // Wywołujemy naszą główną metodę
        });

        aktualizujCene();
    }
        private void aktualizujCene(){
            double cenaBazowa = 0.0;
            RadioGroup radioGroupRozmiar = findViewById(R.id.radioGroupRozmiar);

            // Krok 3.1: Pobierz ID wybranego przycisku
            int wybranyRozmiarId = radioGroupRozmiar.getCheckedRadioButtonId();

            if (wybranyRozmiarId != -1) { // Sprawdzamy, czy cokolwiek jest zaznaczone
                // Krok 3.2: Znajdź ten przycisk i pobierz jego tekst
                RadioButton wybranyPrzycisk = findViewById(wybranyRozmiarId);
                String wybranyTekst = wybranyPrzycisk.getText().toString();

                // Krok 3.3: Użyj switch na pobranym tekście
                switch (wybranyTekst) {
                    case "Mała (20 PLN)":
                        cenaBazowa = 20.0; // Użyj stałych, jeśli je zadeklarowałeś
                        break;
                    case "Średnia (30 PLN)":
                        cenaBazowa = 30.0;
                        break;
                    case "Duża (40 PLN)":
                        cenaBazowa = 40.0;
                        break;
                }
            }

        }
